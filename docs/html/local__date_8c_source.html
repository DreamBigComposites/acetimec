77 uint8_t atc_local_date_day_of_week(int16_t year, uint8_t month, uint8_t day)
78 {
79  // The "year" starts in March to shift leap year calculation to end.
80  int16_t y = year - (month < 3);
81  int16_t d = y + y/4 - y/100 + y/400 + atc_days_of_week[month-1] + day;
82 
83  // 2000-01-01 was a Saturday=6, so set the offsets accordingly
84  return (d < -1) ? (d + 1) % 7 + 8 : (d + 1) % 7 + 1;
85 }

93 int32_t atc_local_date_to_base_epoch_days(
94  int16_t year, uint8_t month, uint8_t day)
95 {
96  int16_t year_prime = year - ((month <= 2) ? 1 : 0); // year begins on Mar 1
97  uint16_t era = year_prime / 400; // [0,24]
98  uint16_t year_of_era = year_prime - 400 * era; // [0,399]
99 
100  uint8_t month_prime = (month <= 2) ? month + 9 : month - 3; // [0,11]
101  uint16_t days_until_month_prime = atc_to_days_until_month_prime(month_prime);
102  uint16_t day_of_year_prime = days_until_month_prime + day - 1; // [0,365]
103  uint32_t day_of_era = (uint32_t) 365 * year_of_era + (year_of_era / 4)
104  - (year_of_era / 100) + day_of_year_prime; // [0,146096]
105 
106  int32_t day_of_epoch_prime = day_of_era + 146097 * era;
107  return day_of_epoch_prime
108  - (2000 / 400) * 146097 /*shift relative to 2000-03-01*/
109  + 60 /*shift relative to 2000-01-01, 2000 is a leap year*/;
110 }

119 void atc_local_date_from_base_epoch_days(
120  int32_t epoch_days,
121  int16_t *year,
122  uint8_t *month,
123  uint8_t *day)
124 {
125  int32_t day_of_epoch_prime = epoch_days + (2000 / 400) * 146097 - 60;
126  uint16_t era = (uint32_t) day_of_epoch_prime / 146097; // [0,24]
127  uint32_t day_of_era = day_of_epoch_prime - 146097 * era; // [0,146096]
128  uint16_t year_of_era = (day_of_era - day_of_era / 1460 + day_of_era / 36524
129  - day_of_era / 146096) / 365; // [0,399]
130  uint16_t year_prime = year_of_era + 400 * era; // [0,9999]
131  uint16_t day_of_year_prime = day_of_era
132  - (365 * year_of_era + year_of_era / 4 - year_of_era/100); // [0,365]
133  uint8_t month_prime = (5 * day_of_year_prime + 2) / 153;
134  uint16_t days_until_month_prime = atc_to_days_until_month_prime(month_prime);
135 
136  *day = day_of_year_prime - days_until_month_prime + 1; // [1,31]
137  *month = (month_prime < 10) ? month_prime + 3 : month_prime - 9; // [1,12]
138  *year = year_prime + ((*month <= 2) ? 1 : 0); // [1,9999]
139 }
