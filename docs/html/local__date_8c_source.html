51 uint8_t atc_local_date_day_of_week(int16_t year, uint8_t month, uint8_t day)
52 {
53  // The "year" starts in March to shift leap year calculation to end.
54  int16_t y = year - (month < 3);
55  int16_t d = y + y/4 - y/100 + y/400 + atc_days_of_week[month-1] + day;
56 
57  // 2000-01-01 was a Saturday=6, so set the offsets accordingly
58  return (d < -1) ? (d + 1) % 7 + 8 : (d + 1) % 7 + 1;
59 }

74 int32_t atc_local_date_to_epoch_days(
75  int8_t year_tiny, uint8_t month, uint8_t day)
76 {
77  int8_t year_prime_tiny = year_tiny - ((month <= 2) ? 1 : 0);
78  int8_t era_tiny = (year_prime_tiny < 0) ? -1 : 0;
79  uint8_t month_prime = (month <= 2) ? month + 9 : month - 3;
80  uint16_t days_until_month_prime = atc_to_days_until_month_prime(month_prime);
81  uint16_t day_of_year_prime = days_until_month_prime + day - 1;
82  uint16_t year_of_era = year_prime_tiny + ((era_tiny < 0) ? 400 : 0);
83  uint32_t day_of_era = (uint32_t) 365 * year_of_era + (year_of_era / 4)
84  - (year_of_era / 100) + day_of_year_prime;
85

87  day_of_era - ((era_tiny < 0) ? 146097 : 0);

93 void atc_local_date_from_epoch_days(
94  int32_t epoch_days,
95  int8_t *year_tiny,
96  uint8_t *month,
97  uint8_t *day)
98 {
99 
100  int32_t day_of_ace_time_epoch_prime = epoch_days - 60;
101  int8_t era_tiny = (day_of_ace_time_epoch_prime < 0) ? -1 : 0;
102  uint32_t day_of_era = day_of_ace_time_epoch_prime
103  + ((era_tiny < 0) ? 146097 : 0);
104  uint16_t year_of_era = (day_of_era - day_of_era / 1460 + day_of_era / 36524
105  - ((day_of_era == 146096) ? 1 : 0)) / 365;
106  int8_t year_prime_tiny = year_of_era - ((era_tiny < 0) ? 400 : 0);
107  uint16_t day_of_year_prime = day_of_era
108  - (365 * year_of_era + year_of_era / 4 - year_of_era/100);
109  uint8_t month_prime = (5 * day_of_year_prime + 2) / 153;
110  uint16_t days_until_month_prime = atc_to_days_until_month_prime(month_prime);
111 
112  *day = day_of_year_prime - days_until_month_prime + 1;
113  *month = (month_prime < 10) ? month_prime + 3 : month_prime - 9;
114  *year_tiny = year_prime_tiny + ((*month <= 2) ? 1 : 0);
115 }
