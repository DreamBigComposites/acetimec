52 uint8_t atc_local_date_day_of_week(int16_t year, uint8_t month, uint8_t day)
53 {
54  // The "year" starts in March to shift leap year calculation to end.
55  int16_t y = year - (month < 3);
56  int16_t d = y + y/4 - y/100 + y/400 + atc_days_of_week[month-1] + day;
57 
58  // 2000-01-01 was a Saturday=6, so set the offsets accordingly
59  return (d < -1) ? (d + 1) % 7 + 8 : (d + 1) % 7 + 1;
60 }
