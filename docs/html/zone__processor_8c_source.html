16 int8_t atc_compare_era_to_year_month(
17  const AtcZoneEra *era,
18  int16_t year,
19  uint8_t month)
20 {
21  if (era->until_year < year) return -1;
22  if (era->until_year > year) return 1;
23  if (era->until_month < month) return -1;
24  if (era->until_month > month) return 1;
25  if (era->until_day > 1) return 1;
26  //if (era->until_time_minutes < 0) return -1; // never possible
27  if (era->until_time_code > 0) return 1;
28  return 0;
29 }

60 void atc_create_matching_era(
61  AtcMatchingEra *new_match,
62  AtcMatchingEra *prev_match,
63  const AtcZoneEra *era,
64  AtcYearMonth start_ym,
65  AtcYearMonth until_ym) {
66 
67  // If prev_match is null, set start_date to be earlier than all valid
68  // ZoneEra.
69  AtcDateTuple start_date;
70  if (prev_match == NULL) {
71  start_date.year = kAtcInvalidYear;
72  start_date.month = 1;
73  start_date.day = 1;
74  start_date.minutes = 0;
75  start_date.suffix = kAtcSuffixW;
76  } else {
77  start_date.year = prev_match->era->until_year;
78  start_date.month = prev_match->era->until_month;
79  start_date.day = prev_match->era->until_day;
80  start_date.minutes = atc_zone_era_until_minutes(prev_match->era);
81  start_date.suffix = atc_zone_era_until_suffix(prev_match->era);
82  }
83  AtcDateTuple lower_bound = {
84  start_ym.year,
85  start_ym.month,
86  1,
87  0,
88  kAtcSuffixW
89  };
90  if (atc_date_tuple_compare(&start_date, &lower_bound) < 0) {
91  start_date = lower_bound;
92  }
93 
94  AtcDateTuple until_date = {
95  era->until_year,
96  era->until_month,
97  era->until_day,
98  atc_zone_era_until_minutes(era),
99  atc_zone_era_until_suffix(era),
100  };
101  AtcDateTuple upper_bound = {
102  until_ym.year,
103  until_ym.month,
104  1,
105  0,
106  kAtcSuffixW
107  };
108  if (atc_date_tuple_compare(&upper_bound, &until_date) < 0) {
109  until_date = upper_bound;
110  }
111 
112  new_match->start_dt = start_date;
113  new_match->until_dt = until_date;
114  new_match->era = era;

120 AtcMonthDay atc_processor_calc_start_day_of_month(
121  int16_t year,
122  uint8_t month,
123  uint8_t on_day_of_week,
124  int8_t on_day_of_month)
125 {
126  AtcMonthDay md;
127 
128  if (on_day_of_week == 0) {
129  md.month = month;
130  md.day = on_day_of_month;
131  return md;
132  }
133 
134  if (on_day_of_month >= 0) {
135  uint8_t days_in_month = atc_local_date_days_in_year_month(year, month);
136  if (on_day_of_month == 0) {
137  on_day_of_month = days_in_month - 6;
138  }
139  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
140  uint8_t day_of_week_shift = (on_day_of_week - dow + 7) % 7;
141  uint8_t day = (uint8_t) (on_day_of_month + day_of_week_shift);
142  if (day > days_in_month) {
143  // TODO: Support shifting from Dec to Jan of following year.
144  day -= days_in_month;
145  month++;
146  }
147  md.month = month;
148  md.day = day;
149  } else {
150  on_day_of_month = -on_day_of_month;
151  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
152  int8_t day_of_week_shift = (dow - on_day_of_week + 7) % 7;
153  int8_t day = on_day_of_month - day_of_week_shift;
154  if (day < 1) {
155  // TODO: Support shifting from Jan to Dec of the previous year.
156  month--;
157  uint8_t days_in_prev_month = atc_local_date_days_in_year_month(
158  year, month);
159  day += days_in_prev_month;
160  }
161  md.month = month;
162  md.day = day;
163  }
164  return md;
165 }

171 uint8_t atc_processor_find_matches(
172  const AtcZoneInfo *zone_info,
173  AtcYearMonth start_ym,
174  AtcYearMonth until_ym,
175  AtcMatchingEra *matches,
176  uint8_t num_matches)
177 {
178  uint8_t i_match = 0;
179  AtcMatchingEra *prev_match = NULL;
180  uint8_t num_eras = zone_info->num_eras;
181  for (uint8_t i_era = 0; i_era < num_eras; i_era++) {
182  const AtcZoneEra *era = &zone_info->eras[i_era];
183  if (atc_era_overlaps_interval(
184  (prev_match ? prev_match->era : NULL),
185  era, start_ym, until_ym)) {
186  if (i_match < num_matches) {
187  atc_create_matching_era(
188  &matches[i_match], prev_match, era, start_ym, until_ym);
189  prev_match = &matches[i_match];
190  i_match++;
191  }
192  }
193  }
194  return i_match;
195 }

296 void atc_processor_find_candidate_transitions(
297  AtcTransitionStorage *ts,
298  AtcMatchingEra *match)
299 {
300  const AtcZonePolicy *policy = match->era->zone_policy;
301  uint8_t num_rules = policy->num_rules;
302  int16_t start_year = match->start_dt.year;
303  int16_t end_year = match->until_dt.year;
304 
305  AtcTransition **prior = atc_transition_storage_reserve_prior(ts);
306  (*prior)->is_valid_prior = false;
307  const char* const* letters = ts->zone_info->zone_context->letters;
308  for (uint8_t r = 0; r < num_rules; r++) {
309  const AtcZoneRule *rule = &policy->rules[r];
310 
311  // Add transitions for interior years
312  int16_t interior_years[kAtcMaxInteriorYears];
313  uint8_t num_years = atc_processor_calc_interior_years(
314  interior_years,
315  kAtcMaxInteriorYears,
316  rule->from_year,
317  rule->to_year,
318  start_year,
319  end_year);
320  for (uint8_t y = 0; y < num_years; y++) {
321  int16_t year = interior_years[y];
322  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
323  atc_processor_create_transition_for_year(t, year, rule, match, letters);
324  uint8_t status = atc_transition_compare_to_match_fuzzy(t, match);
325  if (status == kAtcMatchStatusPrior) {
326  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
327  } else if (status == kAtcMatchStatusWithinMatch) {
328  atc_transition_storage_add_free_agent_to_candidate_pool(ts);
329  } else {
330  // Must be kFarFuture.
331  // Do nothing, allowing the free agent to be reused.
332  }
333  }
334 
335  // Add Transition for prior year
336  int16_t prior_year = atc_processor_get_most_recent_prior_year(
337  rule->from_year, rule->to_year,
338  start_year, end_year);
339  if (prior_year != kAtcInvalidYear) {
340  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
341  atc_processor_create_transition_for_year(
342  t, prior_year, rule, match, letters);
343  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
344  }
345  }
346 
347  // Add the reserved prior into the Candidate pool only if 'isValidPrior' is
348  // true.
349  if ((*prior)->is_valid_prior) {
350  atc_transition_storage_add_prior_to_candidate_pool(ts);
351  }
352 }

358 void atc_processor_process_transition_match_status(
359  AtcTransition *transition,
360  AtcTransition **prior)
361 {
362  uint8_t status = atc_transition_compare_to_match(
363  transition, transition->match);
364  transition->match_status = status;
365 
366  if (status == kAtcMatchStatusExactMatch) {
367  if (*prior) {
368  (*prior)->match_status = kAtcMatchStatusFarPast;
369  }
370  (*prior) = transition;
371  } else if (status == kAtcMatchStatusPrior) {
372  if (*prior) {
373  if (atc_date_tuple_compare(
374  &(*prior)->transition_time_u,
375  &transition->transition_time_u) <= 0) {
376  (*prior)->match_status = kAtcMatchStatusFarPast;
377  (*prior) = transition;
378  } else {
379  transition->match_status = kAtcMatchStatusFarPast;
380  }
381  } else {
382  (*prior) = transition;
383  }
384  }
385 }

462 //---------------------------------------------------------------------------
463 // Step 4
464 //---------------------------------------------------------------------------
465 void atc_processor_generate_start_until_times(
466  AtcTransition **begin,
467  AtcTransition **end)
468 {
469  AtcTransition *prev = *begin;
470  bool is_after_first = false;
471 
472  for (AtcTransition **iter = begin; iter != end; ++iter) {
473  AtcTransition * const t = *iter;
474 
475  // 1) Update the untilDateTime of the previous Transition
476  const AtcDateTuple *tt = &t->transition_time;
477  if (is_after_first) {
478  prev->until_dt = *tt;
479  }
480 
481  // 2) Calculate the current startDateTime by shifting the
482  // transitionTime (represented in the UTC offset of the previous
483  // transition) into the UTC offset of the *current* transition.
484  int16_t minutes = tt->minutes + (
485  - prev->offset_minutes - prev->delta_minutes
486  + t->offset_minutes + t->delta_minutes);
487  t->start_dt.year = tt->year;
488  t->start_dt.month = tt->month;
489  t->start_dt.day = tt->day;
490  t->start_dt.minutes = minutes;
491  t->start_dt.suffix = tt->suffix;
492  atc_date_tuple_normalize(&t->start_dt);
493 
494  // 3) The epochSecond of the 'transitionTime' is determined by the
495  // UTC offset of the *previous* Transition. However, the
496  // transitionTime can be represented by an illegal time (e.g. 24:00).
497  // So, it is better to use the properly normalized startDateTime
498  // (calculated above) with the *current* UTC offset.
499  //
500  // NOTE: We should also be able to calculate this directly from
501  // 'transitionTimeU' which should still be a valid field, because it
502  // hasn't been clobbered by 'untilDateTime' yet. Not sure if this saves
503  // any CPU time though, since we still need to mutiply by 900.
504  const AtcDateTuple *st = &t->start_dt;
505  const atc_time_t offset_seconds = (atc_time_t) 60
506  * (st->minutes - (t->offset_minutes + t->delta_minutes));
507  int32_t epoch_seconds = (int32_t) 86400 * atc_local_date_to_epoch_days(
508  st->year, st->month, st->day);
509  t->start_epoch_seconds = epoch_seconds + offset_seconds;
510 
511  prev = t;
512  is_after_first = true;
513  }
514 
515  // The last Transition's until time is the until time of the MatchingEra.
516  AtcDateTuple until_time_w;
517  AtcDateTuple until_time_s;
518  AtcDateTuple until_time_u;
519  atc_date_tuple_expand(
520  &prev->match->until_dt,
521  prev->offset_minutes,
522  prev->delta_minutes,
523  &until_time_w,
524  &until_time_s,
525  &until_time_u);
526  prev->until_dt = until_time_w;
527 }

534 void atc_processor_create_abbreviation(
535  char *dest,
536  uint8_t dest_size,
537  const char *format,
538  int16_t delta_minutes,
539  const char *letter_string) {
540 
541  // Check if FORMAT contains a '%'.
542  if (strchr(format, '%') != NULL) {
543  // Check if RULES column empty, therefore no 'letter'
544  if (letter_string == NULL) {
545  strncpy(dest, format, dest_size - 1);
546  dest[dest_size - 1] = '\0';
547  } else {
548  atc_copy_replace_string(
549  dest, dest_size, format, '%', letter_string);
550  }
551  } else {
552  // Check if FORMAT contains a '/'.
553  const char* slash_pos = strchr(format, '/');
554  if (slash_pos != NULL) {
555  if (delta_minutes == 0) {
556  uint8_t head_length = (slash_pos - format);
557  if (head_length >= dest_size) head_length = dest_size - 1;
558  memcpy(dest, format, head_length);
559  dest[head_length] = '\0';
560  } else {
561  uint8_t tail_length = strlen(slash_pos+1);
562  if (tail_length >= dest_size) tail_length = dest_size - 1;
563  memcpy(dest, slash_pos+1, tail_length);
564  dest[tail_length] = '\0';
565  }
566  } else {
567  // Just copy the FORMAT disregarding delta_minutes and letter_string.
568  strncpy(dest, format, dest_size);
569  dest[dest_size - 1] = '\0';
570  }
571  }
572 }

607 int8_t atc_processor_init_for_year(
608  AtcZoneProcessor *processor,
609  const AtcZoneInfo *zone_info,
610  int16_t year)
611 {
612  if (processor->zone_info != zone_info) {
613  atc_processor_init(processor);
614  processor->zone_info = zone_info;
615  }
616  if (atc_processor_is_filled_for_year(processor, year)) return kAtcErrOk;
617 
618  processor->year = year;
619  processor->num_matches = 0;
620  atc_transition_storage_init(&processor->transition_storage, zone_info);
621  const AtcZoneContext *context = processor->zone_info->zone_context;
622  if (year < context->start_year - 1 || context->until_year < year) {
623  return kAtcErrGeneric;
624  }
625  AtcYearMonth start_ym = { year - 1, 12 };
626  AtcYearMonth until_ym = { year + 1, 2 };
627 
628  // Step 1: Find matches.
629  uint8_t num_matches = atc_processor_find_matches(
630  processor->zone_info,
631  start_ym,
632  until_ym,
633  processor->matches,
634  kAtcMaxMatches);
635 
636  // Step 2: Create Transitions.
637  atc_processor_create_transitions(
638  &processor->transition_storage,
639  processor->matches,
640  num_matches);
641 
642  // Step 3: Fix transition times of active transitions.
643  AtcTransitionStorage *ts = &processor->transition_storage;
644  AtcTransition **begin = &ts->transitions[0];
645  AtcTransition **end = &ts->transitions[ts->index_prior];
646  atc_transition_fix_times(begin, end);
647 
648  // Step 4: Generate start and until times.
649  atc_processor_generate_start_until_times(begin, end);
650 
651  // Step 5: Calc abbreviations.
652  atc_processor_calc_abbreviations(begin, end);
653 
654  return kAtcErrOk;
655 }

680  const AtcZoneInfo *zone_info,
681  atc_time_t epoch_seconds,

685  processor, zone_info, epoch_seconds);
686  if (err) return err;
687 
688  AtcTransitionForSeconds tfs = atc_transition_storage_find_for_seconds(

692

694  result->dst_offset_minutes = t->delta_minutes;

697  result->abbrev = t->abbrev;
698  result->fold = tfs.fold;
699  if (tfs.num == 2) {

701  } else {
702  result->type = kAtcFindResultExact;

707 // Adapted from ExtendedZoneProcessor::findByLocalDateTime() in the AceTime
708 // library.
709 int8_t atc_processor_find_by_local_date_time(
710  AtcZoneProcessor *processor,
711  const AtcZoneInfo *zone_info,
712  const AtcLocalDateTime *ldt,
713  AtcFindResult *result) {
714 
715  int8_t err = atc_processor_init_for_year(processor, zone_info, ldt->year);
716  if (err) return err;
717 
718  AtcTransitionForDateTime tfd = atc_transition_storage_find_for_date_time(
719  &processor->transition_storage, ldt);
720 
721  // Extract the target Transition, depending on the requested fold
722  // and the tfd.num.
723  const AtcTransition *transition;
724  if (tfd.num == 1) {
725  transition = tfd.curr;
726  result->type = kAtcFindResultExact;
727  result->fold = 0;
728  result->req_std_offset_minutes = transition->offset_minutes;
729  result->req_dst_offset_minutes = transition->delta_minutes;
730  } else { // num = 0 or 2
731  if (tfd.prev == NULL || tfd.curr == NULL) {
732  // ldt was far past or far future
733  transition = NULL;
734  result->type = kAtcFindResultNotFound;
735  result->fold = 0;
736  } else { // gap or overlap
737  if (tfd.num == 0) { // num==0, Gap
738  result->type = kAtcFindResultGap;
739  result->fold = 0;
740  if (ldt->fold == 0) {
741  // ldt wants to use the 'prev' transition to convert to
742  // epochSeconds.
743  result->req_std_offset_minutes = tfd.prev->offset_minutes;
744  result->req_dst_offset_minutes = tfd.prev->delta_minutes;
745  // But after normalization, it will be shifted into the curr
746  // transition, so select 'curr' as the target transition.
747  transition = tfd.curr;
748  } else {
749  // ldt wants to use the 'curr' transition to convert to
750  // epochSeconds.
751  result->req_std_offset_minutes = tfd.curr->offset_minutes;
752  result->req_dst_offset_minutes = tfd.curr->delta_minutes;
753  // But after normalization, it will be shifted into the prev
754  // transition, so select 'prev' as the target transition.
755  transition = tfd.prev;
756  }
757  } else { // num==2, Overlap
758  transition = (ldt->fold == 0) ? tfd.prev : tfd.curr;
759  result->type = kAtcFindResultOverlap;
760  result->fold = ldt->fold;
761  result->req_std_offset_minutes = transition->offset_minutes;
762  result->req_dst_offset_minutes = transition->delta_minutes;
763  }
764  }
765  }
766 
767  if (! transition) {
768  result->type = kAtcFindResultNotFound;
769  return kAtcErrGeneric; // TOOD: should this be kAtcErrOk?
770  }
771 
772  result->std_offset_minutes = transition->offset_minutes;
773  result->dst_offset_minutes = transition->delta_minutes;
774  result->abbrev = transition->abbrev;
775 
776  return kAtcErrOk;
777 }
