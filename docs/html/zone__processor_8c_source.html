19 int8_t atc_compare_era_to_year_month(
20  const AtcZoneEra *era,
21  int16_t year,
22  uint8_t month)
23 {
24  if (era->until_year < year) return -1;
25  if (era->until_year > year) return 1;
26  if (era->until_month < month) return -1;
27  if (era->until_month > month) return 1;
28  if (era->until_day > 1) return 1;
29 
30  int32_t until_time_seconds = atc_zone_era_until_seconds(era);
31  // if (until_time_seconds < 0) return -1; // never possible
32  if (until_time_seconds > 0) return 1;
33  return 0;
34 }

79 void atc_create_matching_era(
80  AtcMatchingEra *new_match,
81  AtcMatchingEra *prev_match,
82  const AtcZoneEra *era,
83  AtcYearMonth start_ym,
84  AtcYearMonth until_ym) {
85 
86  // If prev_match is null, set start_date to be earlier than all valid
87  // ZoneEra.
88  AtcDateTuple start_date;
89  if (prev_match == NULL) {
90  start_date.year = kAtcInvalidYear;
91  start_date.month = 1;
92  start_date.day = 1;
93  start_date.seconds = 0;
94  start_date.suffix = kAtcSuffixW;
95  } else {
96  start_date.year = prev_match->era->until_year;
97  start_date.month = prev_match->era->until_month;
98  start_date.day = prev_match->era->until_day;
99  start_date.seconds = atc_zone_era_until_seconds(prev_match->era);
100  start_date.suffix = atc_zone_era_until_suffix(prev_match->era);
101  }
102  AtcDateTuple lower_bound = {
103  start_ym.year,
104  start_ym.month,
105  1,
106  0,
107  kAtcSuffixW
108  };
109  if (atc_date_tuple_compare(&start_date, &lower_bound) < 0) {
110  start_date = lower_bound;
111  }
112 
113  AtcDateTuple until_date = {
114  era->until_year,
115  era->until_month,
116  era->until_day,
117  atc_zone_era_until_seconds(era),
118  atc_zone_era_until_suffix(era),
119  };
120  AtcDateTuple upper_bound = {
121  until_ym.year,
122  until_ym.month,
123  1,
124  0,
125  kAtcSuffixW
126  };
127  if (atc_date_tuple_compare(&upper_bound, &until_date) < 0) {
128  until_date = upper_bound;
129  }
130 
131  new_match->start_dt = start_date;
132  new_match->until_dt = until_date;
133  new_match->era = era;
134  new_match->prev_match = prev_match;
135  new_match->last_offset_seconds = 0;
136  new_match->last_delta_seconds = 0;
137 }

139 AtcMonthDay atc_processor_calc_start_day_of_month(
140  int16_t year,
141  uint8_t month,
142  uint8_t on_day_of_week,
143  int8_t on_day_of_month)
144 {
145  AtcMonthDay md;
146 
147  if (on_day_of_week == 0) {
148  md.month = month;
149  md.day = on_day_of_month;
150  return md;
151  }
152 
153  if (on_day_of_month >= 0) {
154  uint8_t days_in_month = atc_local_date_days_in_year_month(year, month);
155  if (on_day_of_month == 0) {
156  on_day_of_month = days_in_month - 6;
157  }
158  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
159  uint8_t day_of_week_shift = (on_day_of_week - dow + 7) % 7;
160  uint8_t day = (uint8_t) (on_day_of_month + day_of_week_shift);
161  if (day > days_in_month) {
162  // TODO: Support shifting from Dec to Jan of following year.
163  day -= days_in_month;
164  month++;
165  }
166  md.month = month;
167  md.day = day;
168  } else {
169  on_day_of_month = -on_day_of_month;
170  uint8_t dow = atc_local_date_day_of_week(year, month, on_day_of_month);
171  int8_t day_of_week_shift = (dow - on_day_of_week + 7) % 7;
172  int8_t day = on_day_of_month - day_of_week_shift;
173  if (day < 1) {
174  // TODO: Support shifting from Jan to Dec of the previous year.
175  month--;
176  uint8_t days_in_prev_month = atc_local_date_days_in_year_month(
177  year, month);
178  day += days_in_prev_month;
179  }
180  md.month = month;
181  md.day = day;
182  }
183  return md;
184 }

190 uint8_t atc_processor_find_matches(
191  const AtcZoneInfo *zone_info,
192  AtcYearMonth start_ym,
193  AtcYearMonth until_ym,
194  AtcMatchingEra *matches,
195  uint8_t matches_size)
196 {
197  uint8_t i_match = 0;
198  AtcMatchingEra *prev_match = NULL;
199  uint8_t num_eras = zone_info->num_eras;
200  for (uint8_t i_era = 0; i_era < num_eras; i_era++) {
201  const AtcZoneEra *era = &zone_info->eras[i_era];
202  if (atc_era_overlaps_interval(
203  (prev_match ? prev_match->era : NULL),
204  era, start_ym, until_ym)) {
205  if (i_match < matches_size) {
206  atc_create_matching_era(
207  &matches[i_match], prev_match, era, start_ym, until_ym);
208  prev_match = &matches[i_match];
209  i_match++;
210  }
211  }
212  }
213  return i_match;
214 }

315 void atc_processor_find_candidate_transitions(
316  AtcTransitionStorage *ts,
317  AtcMatchingEra *match)
318 {
319  const AtcZonePolicy *policy = match->era->zone_policy;
320  uint8_t num_rules = policy->num_rules;
321  int16_t start_year = match->start_dt.year;
322  int16_t end_year = match->until_dt.year;
323 
324  AtcTransition **prior = atc_transition_storage_reserve_prior(ts);
325  (*prior)->is_valid_prior = false;
326  const char* const* letters = ts->zone_info->zone_context->letters;
327  for (uint8_t r = 0; r < num_rules; r++) {
328  const AtcZoneRule *rule = &policy->rules[r];
329 
330  // Add transitions for interior years
331  int16_t interior_years[kAtcMaxInteriorYears];
332  uint8_t num_years = atc_processor_calc_interior_years(
333  interior_years,
334  kAtcMaxInteriorYears,
335  rule->from_year,
336  rule->to_year,
337  start_year,
338  end_year);
339  for (uint8_t y = 0; y < num_years; y++) {
340  int16_t year = interior_years[y];
341  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
342  atc_processor_create_transition_for_year(t, year, rule, match, letters);
343  uint8_t status = atc_transition_compare_to_match_fuzzy(t, match);
344  if (status == kAtcComparePrior) {
345  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
346  } else if (status == kAtcCompareWithinMatch) {
347  atc_transition_storage_add_free_agent_to_candidate_pool(ts);
348  } else {
349  // Must be kFarFuture.
350  // Do nothing, allowing the free agent to be reused.
351  }
352  }
353 
354  // Add Transition for prior year
355  int16_t prior_year = atc_processor_get_most_recent_prior_year(
356  rule->from_year, rule->to_year,
357  start_year, end_year);
358  if (prior_year != kAtcInvalidYear) {
359  AtcTransition *t = atc_transition_storage_get_free_agent(ts);
360  atc_processor_create_transition_for_year(
361  t, prior_year, rule, match, letters);
362  atc_transition_storage_set_free_agent_as_prior_if_valid(ts);
363  }
364  }
365 
366  // Add the reserved prior into the Candidate pool only if 'isValidPrior' is
367  // true.
368  if ((*prior)->is_valid_prior) {
369  atc_transition_storage_add_prior_to_candidate_pool(ts);
370  }
371 }

377 void atc_processor_process_transition_match_status(
378  AtcTransition *transition,
379  AtcTransition **prior)
380 {
381  uint8_t status = atc_transition_compare_to_match(
382  transition, transition->match);
383  transition->match_status = status;
384 
385  if (status == kAtcCompareExactMatch) {
386  if (*prior) {
387  (*prior)->match_status = kAtcCompareFarPast;
388  }
389  (*prior) = transition;
390  } else if (status == kAtcComparePrior) {
391  if (*prior) {
392  if (atc_date_tuple_compare(
393  &(*prior)->transition_time_u,
394  &transition->transition_time_u) <= 0) {
395  (*prior)->match_status = kAtcCompareFarPast;
396  (*prior) = transition;
397  } else {
398  transition->match_status = kAtcCompareFarPast;
399  }
400  } else {
401  (*prior) = transition;
402  }
403  }
404 }

481 //---------------------------------------------------------------------------
482 // Step 4
483 //---------------------------------------------------------------------------
484 void atc_processor_generate_start_until_times(
485  AtcTransition **begin,
486  AtcTransition **end)
487 {
488  AtcTransition *prev = *begin;
489  bool is_after_first = false;
490 
491  for (AtcTransition **iter = begin; iter != end; ++iter) {
492  AtcTransition * const t = *iter;
493 
494  // 1) Update the untilDateTime of the previous Transition
495  const AtcDateTuple *tt = &t->transition_time;
496  if (is_after_first) {
497  prev->until_dt = *tt;
498  }
499 
500  // 2) Calculate the current startDateTime by shifting the
501  // transitionTime (represented in the UTC offset of the previous
502  // transition) into the UTC offset of the *current* transition.
503  int32_t seconds = tt->seconds + (
504  - prev->offset_seconds - prev->delta_seconds
505  + t->offset_seconds + t->delta_seconds);
506  t->start_dt.year = tt->year;
507  t->start_dt.month = tt->month;
508  t->start_dt.day = tt->day;
509  t->start_dt.seconds = seconds;
510  t->start_dt.suffix = tt->suffix;
511  atc_date_tuple_normalize(&t->start_dt);
512 
513  // 3) The epochSecond of the 'transitionTime' is determined by the
514  // UTC offset of the *previous* Transition. However, the
515  // transitionTime can be represented by an illegal time (e.g. 24:00).
516  // So, it is better to use the properly normalized startDateTime
517  // (calculated above) with the *current* UTC offset.
518  //
519  // NOTE: We should also be able to calculate this directly from
520  // 'transitionTimeU' which should still be a valid field, because it
521  // hasn't been clobbered by 'untilDateTime' yet. Not sure if this saves
522  // any CPU time though, since we still need to mutiply by 900.
523  const AtcDateTuple *st = &t->start_dt;
524  const atc_time_t offset_seconds = (atc_time_t)
525  (st->seconds - (t->offset_seconds + t->delta_seconds));
526  atc_time_t epoch_seconds = (atc_time_t) 86400
527  * atc_local_date_to_epoch_days(st->year, st->month, st->day);
528  t->start_epoch_seconds = epoch_seconds + offset_seconds;
529 
530  prev = t;
531  is_after_first = true;
532  }
533 
534  // The last Transition's until time is the until time of the MatchingEra.
535  AtcDateTuple until_time_w;
536  AtcDateTuple until_time_s;
537  AtcDateTuple until_time_u;
538  atc_date_tuple_expand(
539  &prev->match->until_dt,
540  prev->offset_seconds,
541  prev->delta_seconds,
542  &until_time_w,
543  &until_time_s,
544  &until_time_u);
545  prev->until_dt = until_time_w;
546 }

553 void atc_processor_create_abbreviation(
554  char *dest,
555  uint8_t dest_size,
556  const char *format,
557  int32_t delta_seconds,
558  const char *letter_string) {
559 
560  // Check if FORMAT contains a '%'.
561  if (strchr(format, '%') != NULL) {
562  // Check if RULES column empty, therefore no 'letter'
563  if (letter_string == NULL) {
564  strncpy(dest, format, dest_size - 1);
565  dest[dest_size - 1] = '\0';
566  } else {
567  atc_copy_replace_string(
568  dest, dest_size, format, '%', letter_string);
569  }
570  } else {
571  // Check if FORMAT contains a '/'.
572  const char* slash_pos = strchr(format, '/');
573  if (slash_pos != NULL) {
574  if (delta_seconds == 0) {
575  uint8_t head_length = (slash_pos - format);
576  if (head_length >= dest_size) head_length = dest_size - 1;
577  memcpy(dest, format, head_length);
578  dest[head_length] = '\0';
579  } else {
580  uint8_t tail_length = strlen(slash_pos+1);
581  if (tail_length >= dest_size) tail_length = dest_size - 1;
582  memcpy(dest, slash_pos+1, tail_length);
583  dest[tail_length] = '\0';
584  }
585  } else {
586  // Just copy the FORMAT disregarding delta_seconds and letter_string.
587  strncpy(dest, format, dest_size);
588  dest[dest_size - 1] = '\0';
589  }
590  }
591 }

637 int8_t atc_processor_init_for_year(
638  AtcZoneProcessor *processor,
639  int16_t year)
640 {
641  if (year == kAtcInvalidYear) return kAtcErrGeneric;
642  if (atc_processor_is_valid_for_year(processor, year)) return kAtcErrOk;
643 
644  const AtcZoneContext *context = processor->zone_info->zone_context;
645  if (year < context->start_year - 1 || context->until_year < year) {
646  return kAtcErrGeneric;
647  }
648 
649  processor->epoch_year = atc_get_current_epoch_year();
650  processor->year = year;
651  processor->num_matches = 0;
652  atc_transition_storage_init(
653  &processor->transition_storage, processor->zone_info);
654 
655  // Fill transitions over a 14-month window straddling the given year.
656  AtcYearMonth start_ym = { year - 1, 12 };
657  AtcYearMonth until_ym = { year + 1, 2 };
658 
659  // Step 1: Find matches.
660  uint8_t num_matches = atc_processor_find_matches(
661  processor->zone_info,
662  start_ym,
663  until_ym,
664  processor->matches,
665  kAtcMaxMatches);
666 
667  // Step 2: Create Transitions.
668  atc_processor_create_transitions(
669  &processor->transition_storage,
670  processor->matches,
671  num_matches);
672 
673  // Step 3: Fix transition times of active transitions.
674  AtcTransitionStorage *ts = &processor->transition_storage;
675  AtcTransition **begin = &ts->transitions[0];
676  AtcTransition **end = &ts->transitions[ts->index_prior];
677  atc_transition_fix_times(begin, end);
678 
679  // Step 4: Generate start and until times.

735 // Adapted from ExtendedZoneProcessor::findByLocalDateTime() in the AceTime
736 // library.
737 void atc_processor_find_by_local_date_time(
738  AtcZoneProcessor *processor,
739  const AtcLocalDateTime *ldt,
740  AtcFindResult *result)
741 {
742 
743  int8_t err = atc_processor_init_for_year(processor, ldt->year);
744  if (err) {
745  result->type = kAtcFindResultNotFound;
746  return;
747  }
748 
749  AtcTransitionForDateTime tfd = atc_transition_storage_find_for_date_time(
750  &processor->transition_storage, ldt);
751 
752  // Extract the target Transition, depending on the requested fold
753  // and the tfd.num.
754  const AtcTransition *transition;
755  if (tfd.num == 1) {
756  transition = tfd.curr;
757  result->type = kAtcFindResultExact;
758  result->fold = 0;
759  result->req_std_offset_seconds = transition->offset_seconds;
760  result->req_dst_offset_seconds = transition->delta_seconds;
761  } else { // num = 0 or 2
762  if (tfd.prev == NULL || tfd.curr == NULL) {
763  // ldt was far past or far future
764  transition = NULL;
765  result->type = kAtcFindResultNotFound;
766  result->fold = 0;
767  } else { // gap or overlap
768  if (tfd.num == 0) { // num==0, Gap
769  result->type = kAtcFindResultGap;
770  result->fold = 0;
771  if (ldt->fold == 0) {
772  // ldt wants to use the 'prev' transition to convert to
773  // epochSeconds.
774  result->req_std_offset_seconds = tfd.prev->offset_seconds;
775  result->req_dst_offset_seconds = tfd.prev->delta_seconds;
776  // But after normalization, it will be shifted into the curr
777  // transition, so select 'curr' as the target transition.
778  transition = tfd.curr;
779  } else {
780  // ldt wants to use the 'curr' transition to convert to
781  // epochSeconds.
782  result->req_std_offset_seconds = tfd.curr->offset_seconds;
783  result->req_dst_offset_seconds = tfd.curr->delta_seconds;
784  // But after normalization, it will be shifted into the prev
785  // transition, so select 'prev' as the target transition.
786  transition = tfd.prev;
787  }
788  } else { // num==2, Overlap
789  transition = (ldt->fold == 0) ? tfd.prev : tfd.curr;
790  result->type = kAtcFindResultOverlap;
791  result->fold = ldt->fold;
792  result->req_std_offset_seconds = transition->offset_seconds;
793  result->req_dst_offset_seconds = transition->delta_seconds;
794  }
795  }
796  }
797 
798  if (! transition) {
799  result->type = kAtcFindResultNotFound;
800  return;
801  }
802 
803  result->std_offset_seconds = transition->offset_seconds;
804  result->dst_offset_seconds = transition->delta_seconds;
805  result->abbrev = transition->abbrev;
806 }
