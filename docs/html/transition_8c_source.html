122 void atc_transition_storage_add_free_agent_to_candidate_pool(
123  AtcTransitionStorage *ts)
124 {
125  if (ts->index_free >= kAtcTransitionStorageSize) return;
126  for (uint8_t i = ts->index_free; i > ts->index_candidate; i--) {
127  AtcTransition *curr = ts->transitions[i];
128  AtcTransition *prev = ts->transitions[i - 1];
129  if (atc_date_tuple_compare(
130  &curr->transition_time,
131  &prev->transition_time) >= 0) break;
132  ts->transitions[i] = prev;
133  ts->transitions[i - 1] = curr;
134  }
135  ts->index_free++;

144 AtcTransition *atc_transition_storage_add_active_candidates_to_active_pool(
145  AtcTransitionStorage *ts)
146 {
147  // Shift active candidates to the left into the Active pool.
148  uint8_t i_active = ts->index_prior;
149  uint8_t i_candidate = ts->index_candidate;
150  for (; i_candidate < ts->index_free; i_candidate++) {
151  if (is_match_status_active(ts->transitions[i_candidate]->match_status)) {
152  if (i_active != i_candidate) {
153  // Perform swap of pointers to AtcTransition.
154  AtcTransition *tmp = ts->transitions[i_active];
155  ts->transitions[i_active] = ts->transitions[i_candidate];
156  ts->transitions[i_candidate] = tmp;
157  }
158  ++i_active;
159  }
160  }
161 
162  ts->index_prior = i_active;
163  ts->index_candidate = i_active;
164  ts->index_free = i_active;
165 
166  return ts->transitions[i_active - 1];
167 }

191 uint8_t atc_transition_compare_to_match(
192  const AtcTransition *t, const AtcMatchingEra *match)
193 {
194  // Find the previous Match offsets.
195  int32_t prev_match_offset_seconds;
196  int32_t prev_match_delta_seconds;
197  if (match->prev_match) {
198  prev_match_offset_seconds = match->prev_match->last_offset_seconds;
199  prev_match_delta_seconds = match->prev_match->last_delta_seconds;
200  } else {
201  prev_match_offset_seconds = atc_zone_era_std_offset_seconds(match->era);
202  prev_match_delta_seconds = 0;
203  }
204 
205  // Expand start times.
206  AtcDateTuple stw;
207  AtcDateTuple sts;
208  AtcDateTuple stu;
209  atc_date_tuple_expand(
210  &match->start_dt,
211  prev_match_offset_seconds,
212  prev_match_delta_seconds,
213  &stw,
214  &sts,
215  &stu);
216 
217  // Transition times.
218  const AtcDateTuple *ttw = &t->transition_time;
219  const AtcDateTuple *tts = &t->transition_time_s;
220  const AtcDateTuple *ttu = &t->transition_time_u;
221 
222  // Compare Transition to Match, where equality is assumed if *any* of the
223  // 'w', 's', or 'u' versions of the DateTuple are equal. This prevents
224  // duplicate Transition instances from being created in a few cases.
225  if (atc_date_tuple_compare(ttw, &stw) == 0
226  || atc_date_tuple_compare(tts, &sts) == 0
227  || atc_date_tuple_compare(ttu, &stu) == 0) {
228  return kAtcCompareExactMatch;
229  }
230 
231  if (atc_date_tuple_compare(ttu, &stu) < 0) {
232  return kAtcComparePrior;
233  }
234 
235  // Now check if the transition occurs after the given match. The
236  // untilDateTime of the current match uses the same UTC offsets as the
237  // transitionTime of the current transition, so no complicated adjustments
238  // are needed. We just make sure we compare 'w' with 'w', 's' with 's',
239  // and 'u' with 'u'.
240  const AtcDateTuple *match_until = &match->until_dt;
241  const AtcDateTuple *transition_time;
242  if (match_until->suffix == kAtcSuffixS) {
243  transition_time = tts;
244  } else if (match_until->suffix == kAtcSuffixU) {
245  transition_time = ttu;
246  } else { // assume 'w'
247  transition_time = ttw;
248  }
249  if (atc_date_tuple_compare(transition_time, match_until) < 0) {
250  return kAtcCompareWithinMatch;
251  }
252 
253  return kAtcCompareFarFuture;
254 }

267 static void atc_calculate_fold_and_overlap(
268  uint8_t* fold,
269  uint8_t* num,
270  const AtcTransition* prev,
271  const AtcTransition* curr,
272  const AtcTransition* next,
273  atc_time_t epoch_seconds)
274 {
275  if (curr == NULL) {
276  *fold = 0;
277  *num = 0;
278  return;
279  }
280 
281  // Check if within forward overlap shadow from prev
282  bool is_overlap;
283  if (prev == NULL) {
284  is_overlap = false;
285  } else {
286  // Extract the shift from prev transition. Can be 0 in some cases where
287  // the zone changed from DST of one zone to the STD into another zone,
288  // causing the overall UTC offset to remain unchanged.
289  atc_time_t shift_seconds = atc_date_tuple_subtract(
290  &curr->start_dt, &prev->until_dt);
291  if (shift_seconds >= 0) {
292  // spring forward, or unchanged
293  is_overlap = false;
294  } else {
295  is_overlap = epoch_seconds - curr->start_epoch_seconds < -shift_seconds;
296  }
297  }
298  if (is_overlap) {
299  *fold = 1; // selects the second match
300  *num = 2;
301  return;
302  }
303 
304  // Check if within backward overlap shawdow from next
305  if (next == NULL) {
306  is_overlap = false;
307  } else {
308  // Extract the shift to next transition. Can be 0 in some cases where
309  // the zone changed from DST of one zone to the STD into another zone,
310  // causing the overall UTC offset to remain unchanged.
311  atc_time_t shift_seconds = atc_date_tuple_subtract(
312  &next->start_dt, &curr->until_dt);
313  if (shift_seconds >= 0) {
314  // spring forward, or unchanged
315  is_overlap = false;
316  } else {
317  // Check if within the backward overlap shadow from next
318  is_overlap = next->start_epoch_seconds - epoch_seconds <= -shift_seconds;
319  }
320  }
321  if (is_overlap) {
322  *fold = 0; // epochSeconds selects the first match
323  *num = 2;
324  return;
325  }
326 
327  // Normal single match, no overlap.
328  *fold = 0;
329  *num = 1;
330 }

332 AtcTransitionForSeconds atc_transition_storage_find_for_seconds(
333  const AtcTransitionStorage *ts,
334  atc_time_t epoch_seconds)
335 {
336  const AtcTransition *prev = NULL;
337  const AtcTransition *curr = NULL;
338  const AtcTransition *next = NULL;
339  for (uint8_t i = 0; i < ts->index_free; i++) {
340  next = ts->transitions[i];
341  if (next->start_epoch_seconds > epoch_seconds) break;
342  prev = curr;
343  curr = next;
344  next = NULL;
345  }
346 
347  uint8_t fold;
348  uint8_t num;
349  atc_calculate_fold_and_overlap(&fold, &num, prev, curr, next, epoch_seconds);
350  AtcTransitionForSeconds result = {curr, fold, num};
351  return result;
352 }

354 AtcTransitionForDateTime atc_transition_storage_find_for_date_time(
355  const AtcTransitionStorage *ts,
356  const AtcLocalDateTime *ldt)
357 {
358  // Convert LocalDateTime to DateTuple.
359  AtcDateTuple local_dt = {
360  ldt->year,
361  ldt->month,
362  ldt->day,
363  (ldt->hour * (int32_t) 60 + ldt->minute) * 60 + ldt->second,
364  kAtcSuffixW
365  };
366 
367  // Examine adjacent pairs of Transitions, looking for an exact match, gap,
368  // or overlap.
369  const AtcTransition *prev = NULL;
370  const AtcTransition *curr = NULL;
371  uint8_t num = 0;
372  for (uint8_t i = 0; i < ts->index_free; i++) {
373  curr = ts->transitions[i];
374 
375  const AtcDateTuple *start_dt = &curr->start_dt;
376  const AtcDateTuple *until_dt = &curr->until_dt;
377  bool is_exact_match = atc_date_tuple_compare(start_dt, &local_dt) <= 0
378  && atc_date_tuple_compare(&local_dt, until_dt) < 0;
379 
380  if (is_exact_match) {
381  // Check for a previous exact match to detect an overlap.
382  if (num == 1) {
383  num++;
384  break;
385  }
386 
387  // Loop again to detect an overlap.
388  num = 1;
389  } else if (atc_date_tuple_compare(start_dt, &local_dt) > 0) {
390  // Exit loop since no more curr transition.
391  break;
392  }
393 
394  prev = curr;
395 
396  // Set nullptr so that if the loop runs off the end of the list of
397  // Transitions, the current transition is marked as nullptr.
398  curr = NULL;
399  }
400 
401  // If the prev was an exact match, set curr to the same to avoid confusion.
402  if (num == 1) {
403  curr = prev;
404  }
405 
406  AtcTransitionForDateTime result = {prev, curr, num};
407  return result;
408 }
