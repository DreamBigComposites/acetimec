54 // The current implementation looks up the LocalDateTime using
55 // atc_time_zone_offset_date_time_from_local_date_time(). This uses the fold and
56 // ignores the offset_seconds.
57 //
58 // An alternative implementation is to convert zdt into epoch_seconds, then call
59 // atc_time_zone_offset_date_time_from_epoch_seconds() instead. This uses the
60 // offset_seconds, and ignores the fold.
61 //
62 // It's not clear which implementation is better.
63 //
64 int8_t atc_zoned_date_time_normalize(AtcZonedDateTime *zdt)
65 {
66  // Copy the date/time components.
67  AtcLocalDateTime ldt;
68  ldt.year = zdt->year;
69  ldt.month = zdt->month;
70  ldt.day = zdt->day;
71  ldt.hour = zdt->hour;
72  ldt.minute = zdt->minute;
73  ldt.second = zdt->second;
74  ldt.fold = zdt->fold;
75 
76  // ZonedDateTime memory layout must be same as OffsetDateTime.
77  return atc_time_zone_offset_date_time_from_local_date_time(
78  &zdt->tz, &ldt, (AtcOffsetDateTime *) zdt);
79 }
