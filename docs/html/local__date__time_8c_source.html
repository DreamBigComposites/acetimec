25 int8_t atc_local_date_time_from_epoch_seconds(
26  AtcLocalDateTime *ldt,
27  atc_time_t epoch_seconds)
28 {
29  ldt->fold = 0;
30 
31  if (epoch_seconds == kAtcInvalidEpochSeconds) {
32  ldt->year = kAtcInvalidYear;
33  ldt->month = 0;
34  ldt->day = 0;
35  ldt->hour = 0;
36  ldt->minute = 0;
37  ldt->second = 0;
38  return kAtcErrGeneric;
39  }
40 
41  // Integer floor-division towards -infinity
42  int32_t days = (epoch_seconds < 0)
43  ? (epoch_seconds + 1) / 86400 - 1
44  : epoch_seconds / 86400;
45  int32_t seconds = epoch_seconds - 86400 * days;
46 
47  // Extract (year, month day).
48  atc_local_date_from_epoch_days(days, &ldt->year, &ldt->month, &ldt->day);
49 
50  // Extract (hour, minute, second). The compiler will combine the mod (%) and
51  // division (/) operations into a single (dividend, remainder) function call.
52  ldt->second = seconds % 60;
53  uint16_t minutes = seconds / 60;
54  ldt->minute = minutes % 60;
55  ldt->hour = minutes / 60;
56 
57  return kAtcErrOk;
58 }
