36 void atc_local_date_time_from_epoch_seconds(
37  AtcLocalDateTime *ldt,
38  atc_time_t epoch_seconds)
39 {
40  ldt->fold = 0;
41 
42  if (epoch_seconds == kAtcInvalidEpochSeconds) {
43  atc_local_date_time_set_error(ldt);
44  return;
45  }
46 
47  // Integer floor-division towards -infinity
48  int32_t days = (epoch_seconds < 0)
49  ? (epoch_seconds + 1) / 86400 - 1
50  : epoch_seconds / 86400;
51  int32_t seconds = epoch_seconds - 86400 * days;
52 
53  // Extract (year, month day).
54  atc_local_date_from_epoch_days(days, &ldt->year, &ldt->month, &ldt->day);
55 
56  // Extract (hour, minute, second). The compiler will combine the mod (%) and
57  // division (/) operations into a single (dividend, remainder) function call.
58  ldt->second = seconds % 60;
59  uint16_t minutes = seconds / 60;
60  ldt->minute = minutes % 60;
61  ldt->hour = minutes / 60;
62 }

75 void atc_local_date_time_from_unix_seconds(
76  AtcLocalDateTime *ldt,
77  int64_t unix_seconds)
78 {
79  ldt->fold = 0;
80 
81  if (unix_seconds == kAtcInvalidUnixSeconds) {
82  atc_local_date_time_set_error(ldt);
83  return;
84  }
85 
86  // Integer floor-division towards -infinity. Implicitly assume that
87  // unix_seconds/86400 fits inside an int32_t. So the largest/smallest days is
88  // about +/- 2^31, which translates to +/- 5_879_489 years, which I think is
89  // more than sufficient for the foreseeable future.
90  int32_t unix_days = (unix_seconds < 0)
91  ? (unix_seconds + 1) / 86400 - 1
92  : unix_seconds / 86400;
93  int32_t seconds = unix_seconds - 86400 * unix_days;
94 
95  int32_t days = atc_epoch_days_from_unix_days(unix_days);
96 
97  // Extract (year, month day).
98  atc_local_date_from_epoch_days(days, &ldt->year, &ldt->month, &ldt->day);
99 
100  // Extract (hour, minute, second). The compiler will combine the mod (%) and
101  // division (/) operations into a single (dividend, remainder) function call.
102  ldt->second = seconds % 60;
103  uint16_t minutes = seconds / 60;
104  ldt->minute = minutes % 60;
105  ldt->hour = minutes / 60;
106 }
