18 // Adapted from TimeZone::getOffsetDateTime(epochSeconds) from the
19 // AceTime library.
20 void atc_time_zone_offset_date_time_from_epoch_seconds(
21  const AtcTimeZone *tz,
22  atc_time_t epoch_seconds,
23  AtcOffsetDateTime *odt)
24 {
25  int32_t offset_seconds;
26  uint8_t fold;
27  if (tz->zone_info) {
28  atc_processor_init_for_zone_info(tz->zone_processor, tz->zone_info);
29 
30  AtcFindResult result;
31  atc_processor_find_by_epoch_seconds(
32  tz->zone_processor, epoch_seconds, &result);
33  if (result.type == kAtcFindResultNotFound) {
34  atc_offset_date_time_set_error(odt);
35  return;
36  }
37  offset_seconds = result.std_offset_seconds + result.dst_offset_seconds;
38  fold = result.fold;
39  } else {
40  offset_seconds = 0;
41  fold = 0;
42  }
43 
44  atc_offset_date_time_from_epoch_seconds(odt, epoch_seconds, offset_seconds);
45  odt->fold = fold;
46 }

48 // Adapted from TimeZone::getOffsetDateTime(const LocalDatetime&) from the
49 // AceTime library.
50 void atc_time_zone_offset_date_time_from_local_date_time(
51  const AtcTimeZone *tz,
52  const AtcLocalDateTime *ldt,
53  AtcOffsetDateTime *odt)
54 {
55  if (tz->zone_info) {
56  atc_processor_init_for_zone_info(tz->zone_processor, tz->zone_info);
57 
58  AtcFindResult result;
59  atc_processor_find_by_local_date_time(tz->zone_processor, ldt, &result);
60  if (result.type == kAtcFindResultNotFound) {
61  atc_offset_date_time_set_error(odt);
62  return;
63  }
64 
65  // Convert FindResult into OffsetDateTime using the requested offset.
66  odt->year = ldt->year;
67  odt->month = ldt->month;
68  odt->day = ldt->day;
69  odt->hour = ldt->hour;
70  odt->minute = ldt->minute;
71  odt->second = ldt->second;
72  odt->offset_seconds =
73  result.req_std_offset_seconds + result.req_dst_offset_seconds;
74  odt->fold = result.fold;
75 
76  // Special process for kAtcFindResultGap: Convert to epoch_seconds using the
77  // req_std_offset_seconds and req_dst_offset_seconds, then convert back to
78  // AtcOffsetDateTime using the target's std_offset_seconds and
79  // dst_offset_seconds.
80  if (result.type == kAtcFindResultGap) {
81  atc_time_t epoch_seconds = atc_offset_date_time_to_epoch_seconds(odt);
82  int32_t target_offset =
83  result.std_offset_seconds + result.dst_offset_seconds;
84  atc_offset_date_time_from_epoch_seconds(
85  odt, epoch_seconds, target_offset);
86  }
87  } else {
88  odt->year = ldt->year;
89  odt->month = ldt->month;
90  odt->day = ldt->day;
91  odt->hour = ldt->hour;
92  odt->minute = ldt->minute;
93  odt->second = ldt->second;
94  odt->offset_seconds = 0;
95  odt->fold = 0;
96  }
97 }

99 void atc_time_zone_zoned_extra_from_epoch_seconds(
100  const AtcTimeZone *tz,
101  atc_time_t epoch_seconds,
102  AtcZonedExtra *extra)
103 {
104  if (epoch_seconds == kAtcInvalidEpochSeconds) {
105  atc_zoned_extra_set_error(extra);
106  return;
107  }
108 
109  if (tz->zone_info) {
110  atc_processor_init_for_zone_info(tz->zone_processor, tz->zone_info);
111 
112  AtcFindResult result;
113  atc_processor_find_by_epoch_seconds(
114  tz->zone_processor, epoch_seconds, &result);
115  extra->type = result.type;
116  if (result.type == kAtcFindResultNotFound) {
117  return;
118  }
119 
120  extra->std_offset_seconds = result.std_offset_seconds;
121  extra->dst_offset_seconds = result.dst_offset_seconds;
122  extra->req_std_offset_seconds = result.req_std_offset_seconds;
123  extra->req_dst_offset_seconds = result.req_dst_offset_seconds;
124  memcpy(extra->abbrev, result.abbrev, kAtcAbbrevSize);
125  extra->abbrev[kAtcAbbrevSize - 1] = '\0';
126  } else {
127  extra->type = kAtcFindResultExact;
128  extra->std_offset_seconds = 0;
129  extra->dst_offset_seconds = 0;
130  extra->req_std_offset_seconds = 0;
131  extra->req_dst_offset_seconds = 0;
132  memcpy(extra->abbrev, "UTC", sizeof("UTC"));
133  }
134 }
