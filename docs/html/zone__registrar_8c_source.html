37 static uint16_t binary_search(
38  const AtcZoneInfo * const * registry,
39  uint16_t size,
40  uint32_t zone_id)
41 {
42  uint16_t a = 0;
43  uint16_t b = size;
44  while (1) {
45  uint16_t diff = b - a;
46  if (diff == 0) break;
47 
48  uint16_t c = a + diff / 2;
49  const AtcZoneInfo *curr = registry[c];
50  if (curr->zone_id == zone_id) return c;
51  if (zone_id < curr->zone_id) {
52  b = c;
53  } else {
54  a = c + 1;
55  }
56  }
57  return UINT16_MAX;
58 }
