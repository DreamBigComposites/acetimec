13 int8_t atc_date_tuple_compare(
14  const AtcDateTuple *a,
15  const AtcDateTuple *b)
16 {
17  if (a->year < b->year) return -1;
18  if (a->year > b->year) return 1;
19  if (a->month < b->month) return -1;
20  if (a->month > b->month) return 1;
21  if (a->day < b->day) return -1;
22  if (a->day > b->day) return 1;
23  if (a->seconds < b->seconds) return -1;
24  if (a->seconds > b->seconds) return 1;
25  return 0;
26 }

28 atc_time_t atc_date_tuple_subtract(
29  const AtcDateTuple *a,
30  const AtcDateTuple *b)
31 {
32  int32_t da = atc_local_date_to_epoch_days(a->year, a->month, a->day);
33  int32_t db = atc_local_date_to_epoch_days(b->year, b->month, b->day);
34 
35  // Subtract the days, before converting to seconds, to avoid overflowing the
36  // int32_t when a.year or b.year is more than 68 years from the
37  // atc_current_epoch_year.
38  return (da - db) * 86400 + (a->seconds - b->seconds);
39 }

101 void atc_date_tuple_normalize(AtcDateTuple *dt)
102 {
103  const int32_t kOneDayAsSeconds = 60 * 60 * 24;
104 
105  if (dt->seconds <= -kOneDayAsSeconds) {
106  atc_local_date_decrement_one_day(&dt->year, &dt->month, &dt->day);
107  dt->seconds += kOneDayAsSeconds;
108  } else if (kOneDayAsSeconds <= dt->seconds) {
109  atc_local_date_increment_one_day(&dt->year, &dt->month, &dt->day);
110  dt->seconds -= kOneDayAsSeconds;
111  } else {
112  // do nothing
113  }
114 }

116 uint8_t atc_date_tuple_compare_fuzzy(
117  const AtcDateTuple *t,
118  const AtcDateTuple *start,
119  const AtcDateTuple *until)
120 {
121  // Use int32_t because a delta year of 2730 or greater will exceed
122  // the range of an int16_t.
123  int32_t t_months = t->year * (int32_t) 12 + t->month;
124  int32_t start_months = start->year * (int32_t) 12 + start->month;
125  if (t_months < start_months - 1) return kAtcComparePrior;
126  int32_t until_months = until->year * 12 + until->month;
127  if (until_months + 1 < t_months) return kAtcCompareFarFuture;
128  return kAtcCompareWithinMatch;
129 }
