46 int32_t atc_convert_to_days(int16_t year, uint8_t month, uint8_t day)
47 {
48  int16_t year_prime = year - ((month <= 2) ? 1 : 0); // year begins on Mar 1
49  uint16_t era = year_prime / 400; // [0,24]
50  uint16_t year_of_era = year_prime - 400 * era; // [0,399]
51 
52  uint8_t month_prime = (month <= 2) ? month + 9 : month - 3; // [0,11]
53  uint16_t days_until_month_prime =
54  atc_convert_to_days_until_month_prime(month_prime);
55  uint16_t day_of_year_prime = days_until_month_prime + day - 1; // [0,365]
56  uint32_t day_of_era = (uint32_t) 365 * year_of_era + (year_of_era / 4)
57  - (year_of_era / 100) + day_of_year_prime; // [0,146096]
58 
59  // epoch_prime days is relative to 0000-03-01
60  int32_t day_of_epoch_prime = day_of_era + 146097 * era;
61  return day_of_epoch_prime
62  - (kAtcConverterEpochYear / 400) * 146097 // relative to 2000-03-01
63  + 60; // relative to 2000-01-01, 2000 is a leap year
64 }

67  int32_t epoch_days,
68  int16_t *year,
69  uint8_t *month,
70  uint8_t *day)
71 {
72  // epoch_prime days is relative to 0000-03-01
73  int32_t day_of_epoch_prime = epoch_days
74  + (kAtcConverterEpochYear / 400) * 146097
75  - 60;
76 
77  uint16_t era = (uint32_t) day_of_epoch_prime / 146097; // [0,24]
78  uint32_t day_of_era = day_of_epoch_prime - 146097 * era; // [0,146096]
79  uint16_t year_of_era = (day_of_era - day_of_era / 1460 + day_of_era / 36524
80  - day_of_era / 146096) / 365; // [0,399]
81  uint16_t year_prime = year_of_era + 400 * era; // [0,9999]
82  uint16_t day_of_year_prime = day_of_era
83  - (365 * year_of_era + year_of_era / 4 - year_of_era/100); // [0,365]
84  uint8_t month_prime = (5 * day_of_year_prime + 2) / 153;
85  uint16_t days_until_month_prime =
86  atc_convert_to_days_until_month_prime(month_prime);
87 
88  *day = day_of_year_prime - days_until_month_prime + 1; // [1,31]
89  *month = (month_prime < 10) ? month_prime + 3 : month_prime - 9; // [1,12]
90  *year = year_prime + ((*month <= 2) ? 1 : 0); // [1,9999]
91 }
