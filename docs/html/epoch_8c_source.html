85 // See AceTime/src/ace_time/internal/EpochConverterHinnant.h.
86 int32_t atc_convert_to_internal_days(int16_t year, uint8_t month, uint8_t day)
87 {
88  int16_t year_prime = year - ((month <= 2) ? 1 : 0); // year begins on Mar 1
89  uint16_t era = year_prime / 400; // [0,24]
90  uint16_t year_of_era = year_prime - 400 * era; // [0,399]
91 
92  uint8_t month_prime = (month <= 2) ? month + 9 : month - 3; // [0,11]
93  uint16_t days_until_month_prime =
94  atc_convert_to_days_until_month_prime(month_prime);
95  uint16_t day_of_year_prime = days_until_month_prime + day - 1; // [0,365]
96  uint32_t day_of_era = (uint32_t) 365 * year_of_era + (year_of_era / 4)
97  - (year_of_era / 100) + day_of_year_prime; // [0,146096]
98 
99  // epoch_prime days is relative to 0000-03-01
100  int32_t day_of_epoch_prime = day_of_era + 146097 * era;
101  return day_of_epoch_prime
102  - (kAtcInternalEpochYear / 400) * 146097 // relative to 2000-03-01
103  + 60; // relative to 2000-01-01, 2000 is a leap year
104 }

106 // See AceTime/src/ace_time/internal/EpochConverterHinnant.h.
107 void atc_convert_from_internal_days(
108  int32_t internal_days,
109  int16_t *year,
110  uint8_t *month,
111  uint8_t *day)
112 {
113  // epoch_prime days is relative to 0000-03-01
114  int32_t day_of_epoch_prime = internal_days
115  + (kAtcInternalEpochYear / 400) * 146097
116  - 60;
117 
118  uint16_t era = (uint32_t) day_of_epoch_prime / 146097; // [0,24]
119  uint32_t day_of_era = day_of_epoch_prime - 146097 * era; // [0,146096]
120  uint16_t year_of_era = (day_of_era - day_of_era / 1460 + day_of_era / 36524
121  - day_of_era / 146096) / 365; // [0,399]
122  uint16_t year_prime = year_of_era + 400 * era; // [0,9999]
123  uint16_t day_of_year_prime = day_of_era
124  - (365 * year_of_era + year_of_era / 4 - year_of_era/100); // [0,365]
125  uint8_t month_prime = (5 * day_of_year_prime + 2) / 153;
126  uint16_t days_until_month_prime =
127  atc_convert_to_days_until_month_prime(month_prime);
128 
129  *day = day_of_year_prime - days_until_month_prime + 1; // [1,31]
130  *month = (month_prime < 10) ? month_prime + 3 : month_prime - 9; // [1,12]
131  *year = year_prime + ((*month <= 2) ? 1 : 0); // [1,9999]
132 }
